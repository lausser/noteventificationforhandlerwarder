# Create a key pair for the client

from pynostr.key import PrivateKey
pk = PrivateKey()
print("nsec:", pk.bech32())
print("npub:", pk.public_key.bech32())


# Then use these keys when you send an event

OMD_SITE=my_devel_site \
notificationforwarder \
  --forwarder nostr \
  --formatter nostr \
  --forwarderopt relays='wss://relay.damus.io,wss://nostr-pub.wellorder.net' \
  --forwarderopt nsec='nsec1yourprivatekeyhere' \
  --forwarderopt tags='[["p","npub1targetpubkeyhere"]]' \
  --eventopt HOSTNAME='demo-host' \
  --eventopt SERVICEDESC='cpu' \
